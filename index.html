Code example


const SIZE = 200;
const STEP = 5;
const INTERVAL = 10;
const logo = document.getElementById("logo");

let x = 0;
let y = 0;
let stepX = STEP;
let stepY = STEP;

const onInterval = () => {
  x += stepX;
  y += stepY;
  logo.style.left = `${x}px`;
  logo.style.top = `${y}px`;

  if (y >= window.innerHeight - SIZE || y <= 0) {
    stepY = -stepY;
  }

  if (x >= window.innerWidth - SIZE || x <= 0) {
    stepX = -stepX;
  }
};

setInterval(onInterval, INTERVAL);
